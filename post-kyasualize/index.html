    $kyasual-background: #fffae6;
$kyasual-text: #3c372d;
$kyasual-block-title: #462d28;
$kyasual-block-body: #ebdcb9;
$kyasual-example-title: #5a7323;
$kyasual-example-body: #e1f0c8;
$kyasual-alerted-title: #732346;
$kyasual-alerted-body: #e5c3be;
